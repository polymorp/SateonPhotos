create procedure [dbo].[sproc_FindUnprocessedImages]
as
SELECT o.employeeNo,o.image,o.dataFormat
  FROM [OriginalPhotos] o
  left join [ProcessedPhotos] p on o.employeeNo = p.employeeNo
  where p.employeeNo is null
